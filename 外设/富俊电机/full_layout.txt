{

        unsigned char uchCRCHi = 0xFF ;
        unsigned char uchCRCLo = 0xFF ;
        unsigned uIndex ;
        while (usDataLen--)
        {

                 uIndex = uchCRCHi ^ *(puchMsg++) ;
                 uchCRCHi = uchCRCLo ^ u8_CrcTabHi[uIndex] ;
                 uchCRCLo = u8_CrcTabLo[uIndex] ;
        }
        return (uchCRCHi << 8 | uchCRCLo) ;
}
